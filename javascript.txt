import java.io.IOException;
import java.io.PrintWriter;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;



public class serv1 extends HttpServlet {


    protected void processRequest(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        response.setContentType("text/html;charset=UTF-8");
        try (PrintWriter out = response.getWriter()) {
            String name = request.getParameter("txtname");
            String r = request.getParameter("radiogen");
            String[] languages = {
                request.getParameter("c1"),
                request.getParameter("c2"),
                request.getParameter("c3")};
            String mark = request.getParameter("per");

            out.println("<html>");
            out.println("<body>");
            out.println("<center>");
            out.println("<br>BIO-DATA<br><br>");
            out.println("Name of the student: " + name+ " <br>");
            out.println("Gender: " + r + " <br>");
            out.println("Languages Known: ");


            for (String language : languages) {
                if (language!=null) {
                    out.println(language+" ");
                }
            }
            out.println("<br>");
            out.println("Percentage of marks: " + mark + "<br>");
            out.println("</center>");
            out.println("</body>");
            out.println("</html>");

        }
    }

  
}

import java.io.IOException;
import java.io.PrintWriter;
import java.util.Date;
import javax.servlet.ServletException;
import javax.servlet.http.Cookie;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;


public class serv extends HttpServlet {


    protected void processRequest(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        response.setContentType("text/html;charset=UTF-8");
        try (PrintWriter out = response.getWriter()) {
            Date d =new Date();
            String s = request.getParameter("txtname");
            Cookie newcookie = new Cookie(s, d.toString());
            
            response.addCookie(newcookie);

            // Retrieve and display cookies
            Cookie[] cookies = request.getCookies();
            
          
               for (Cookie cooky : cookies) {
                 out.println(cooky.getName() + " " + cooky.getValue() + "<br>");
            }
            

            out.println("</html>");
        }
    }

    

}
